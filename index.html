What is Swapping?

Swapping means exchanging the values of two variables. A third variable (temp) acts as a temporary holder so no value gets overwritten before it's saved.

swap.c 
#include <stdio.h>   /* Standard Input/Output library */

int main() {
    int a = 10, b = 20, temp;  /* Declare two numbers and a helper variable */

    /* Step 1: Save value of 'a' into temp so we don't lose it */
    temp = a;

    /* Step 2: Overwrite 'a' with value of 'b' */
    a = b;

    /* Step 3: Put the saved value (was 'a') into 'b' */
    b = temp;

    printf("a = %d\n", a);   /* a is now 20 */
    printf("b = %d\n", b);   /* b is now 10 */

    return 0;   /* Exit program successfully */
}

What is ASCII?

ASCII (American Standard Code for Information Interchange) assigns a numeric code to every character — 'A' = 65, 'a' = 97, '0' = 48. Printing a char with %d reveals its numeric code.

ascii.c 
#include <stdio.h>   /* Needed for printf and scanf */

int main() {
    char ch;   /* Variable to store a single character */

    printf("Enter a character: ");
    scanf("%c", &ch);   /* Read one character; &ch is its memory address */

    /* When a char variable is printed with %d, C shows its ASCII number */
    printf("ASCII value of '%c' = %d\n", ch, ch);

    return 0;
}

What is a Leap Year?

A leap year has 366 days. The rule: divisible by 4 AND not by 100 → leap year. Exception: if divisible by 400, it's still a leap year (e.g. year 2000). The % operator gives the remainder.

leapyear.c 
#include <stdio.h>   /* Needed for printf and scanf */

int main() {
    int year;   /* Stores the year entered by the user */

    printf("Enter year: ");
    scanf("%d", &year);

    /* Leap year logic:
       Condition 1: Divisible by 4 but NOT by 100  (e.g., 2024)
       Condition 2: Divisible by 400               (e.g., 2000)
       If either condition is true → it's a leap year */
    if ((year % 4 == 0 && year % 100 != 0) || year % 400 == 0)
        printf("%d is a Leap Year\n", year);
    else
        printf("%d is NOT a Leap Year\n", year);

    return 0;
}

Break vs Continue

break — immediately exits the entire loop. continue — skips only the current iteration and jumps to the next. This program prints 1–7 but skips 5 and stops before 8.

break_continue.c 
#include <stdio.h>   /* Needed for printf */

int main() {
    int i;   /* Loop counter variable */

    /* Loop runs from 1 to 10 */
    for (i = 1; i <= 10; i++) {

        /* 'continue' skips printing when i = 5 and goes to i = 6 */
        if (i == 5)
            continue;

        /* 'break' exits the loop completely when i = 8 */
        if (i == 8)
            break;

        printf("%d\n", i);   /* Output: 1 2 3 4 6 7 (skips 5, stops at 8) */
    }

    return 0;
}

What is a Prime Number?

A prime number is only divisible by 1 and itself (e.g., 2, 3, 5, 7, 11). We use a flag variable: 0 means prime, 1 means not prime. We check divisibility for every number from 2 to n−1.

prime.c 
#include <stdio.h>   /* Needed for printf and scanf */

int main() {
    int n, i, flag = 0;
    /* n     = number to check
       i     = loop counter
       flag  = 0 means prime, 1 means NOT prime */

    printf("Enter a number: ");
    scanf("%d", &n);

    /* Check if any number from 2 to n-1 divides n evenly */
    for (i = 2; i < n; i++) {
        if (n % i == 0) {   /* If remainder is 0, a divisor was found */
            flag = 1;        /* Mark as NOT prime */
            break;           /* No need to check further */
        }
    }

    /* If flag is still 0, no divisor was found → it's prime */
    if (flag == 0)
        printf("%d is a Prime Number\n", n);
    else
        printf("%d is NOT a Prime Number\n", n);

    return 0;
}

What is Call By Reference?

Instead of sending a copy of a variable, we pass its memory address using &. Inside the function, * is used to access and modify the actual value at that address — so changes persist outside the function.

call_by_ref.c 
#include <stdio.h>   /* Needed for printf */

/* Function receives POINTERS (addresses) instead of copies */
void swap(int *a, int *b) {
    int temp;
    temp = *a;   /* *a means: read the value stored at address 'a' */
    *a   = *b;   /* Write value from address b into address a */
    *b   = temp; /* Write saved value into address b */
}

int main() {
    int x = 10, y = 20;

    /* &x and &y give the memory addresses of x and y */
    swap(&x, &y);

    printf("x = %d\n", x);   /* x is now 20 — changed by the function! */
    printf("y = %d\n", y);   /* y is now 10 */

    return 0;
}

What is Recursion?

A recursive function calls itself until a base case stops it. For factorial: 5! = 5 × 4 × 3 × 2 × 1. The base case is 0! = 1. Without a base case the function would loop forever.

factorial.c 
#include <stdio.h>   /* Needed for printf and scanf */

/* Recursive function: keeps calling itself with smaller n */
int factorial(int n) {
    if (n == 0)
        return 1;                     /* Base case: 0! = 1 (stops recursion) */
    else
        return n * factorial(n - 1);  /* Recursive case: n! = n × (n-1)! */
}
/* Call trace for n=4:
   factorial(4) → 4 × factorial(3)
                       3 × factorial(2)
                             2 × factorial(1)
                                   1 × factorial(0) → returns 1
   Result = 4 × 3 × 2 × 1 = 24 */

int main() {
    int num;

    printf("Enter a number: ");
    scanf("%d", &num);

    printf("Factorial of %d = %d\n", num, factorial(num));

    return 0;
}

What are Standard Library Functions?

C comes with ready-made functions in header files: <math.h> gives sqrt() and pow(), while <string.h> gives strlen(). You just #include the header and call them.

library.c 
#include <stdio.h>    /* For printf */
#include <math.h>     /* For sqrt() and pow() functions */
#include <string.h>   /* For strlen() function */

int main() {
    char name[] = "CQUEST";   /* A string (array of characters, ends with '\0') */

    /* sqrt(25) = 5.00 — square root of 25 */
    printf("Square Root of 25 = %.2f\n", sqrt(25));

    /* pow(2, 3) = 8.00 — 2 raised to the power of 3 */
    printf("2 to the power 3 = %.2f\n", pow(2, 3));

    /* strlen counts characters in the string (not including the null '\0') */
    printf("Length of \"%s\" = %lu\n", name, strlen(name));  /* Output: 6 */

    return 0;
}

What is a Matrix?

A matrix is a 2D array: rows and columns of numbers. In C, a[2][2] means 2 rows and 2 columns. Addition/subtraction works element-by-element: c[i][j] = a[i][j] + b[i][j]. Nested for loops handle both dimensions.

matrix.c 
#include <stdio.h>   /* Needed for printf */

int main() {
    /* Two 2×2 matrices (2 rows, 2 columns each) */
    int a[2][2] = {{1, 2}, {3, 4}};
    int b[2][2] = {{5, 6}, {7, 8}};
    int c[2][2];    /* Result matrix (uninitialized, values set below) */
    int i, j;       /* Loop variables: i = row index, j = column index */

    printf("Matrix Addition Result:\n");

    /* Outer loop iterates over rows (i = 0, 1) */
    for (i = 0; i < 2; i++) {

        /* Inner loop iterates over columns (j = 0, 1) */
        for (j = 0; j < 2; j++) {
            c[i][j] = a[i][j] + b[i][j];  /* Add matching elements */
            printf("%d ", c[i][j]);
        }
        printf("\n");  /* Move to next line after each row */
    }
    /* Output:
       6  8
       10 12  */

    return 0;
}

What is Dynamic Memory Allocation?

DMA lets you request memory at runtime instead of compile time. malloc(n) allocates n bytes (uninitialized), calloc(n, size) allocates and zeros the memory. Always call free() when done to prevent memory leaks.

dma.c 
#include <stdio.h>    /* For printf */
#include <stdlib.h>   /* For malloc(), calloc(), and free() */

int main() {
    int *ptr;   /* Pointer to hold the address of allocated memory */
    int i;      /* Loop variable */

    /* malloc allocates space for 3 integers at runtime
       sizeof(int) gives the byte size of one integer (usually 4 bytes)
       The (int*) cast converts the void* return type to int* */
    ptr = (int*) malloc(3 * sizeof(int));

    /* Always check if malloc returned NULL (allocation failure) */
    if (ptr == NULL) {
        printf("Memory allocation failed!\n");
        return 1;
    }

    /* Use the allocated memory like a regular array */
    ptr[0] = 10;
    ptr[1] = 20;
    ptr[2] = 30;

    /* Print stored values */
    for (i = 0; i < 3; i++)
        printf("ptr[%d] = %d\n", i, ptr[i]);

    free(ptr);   /* CRITICAL: Release memory back to the OS to avoid leaks */
    ptr = NULL;  /* Good practice: set pointer to NULL after freeing */

    return 0;
}
